from dataclasses import dataclass, field
from typing import Optional
import datetime


# ─────────────────────────────────────────────
# Data Classes
# ─────────────────────────────────────────────

@dataclass
class MemoryPoolStats:
    name: str
    used_mb: float
    committed_mb: float
    max_mb: float


@dataclass
class BufferPoolStats:
    name: str
    count: int
    used_mb: float
    capacity_mb: float


@dataclass
class ThreadStats:
    thread_count: int
    stack_size_kb: int = 512

    @property
    def estimated_mb(self) -> float:
        return round(self.thread_count * self.stack_size_kb / 1024, 2)


@dataclass
class OffHeapConfig:
    enabled: bool
    size_mb: float


@dataclass
class HeapStats:
    used_mb: float
    committed_mb: float
    max_mb: float


@dataclass
class OverheadSnapshot:
    """Complete overhead memory snapshot for one JVM (driver or executor)."""
    label:           str
    timestamp:       str
    executor_number: Optional[int]       = None
    executor_url:    Optional[str]       = None
    driver_name:     Optional[str]       = None
    memory_pools:    list[MemoryPoolStats]   = field(default_factory=list)
    buffer_pools:    list[BufferPoolStats]   = field(default_factory=list)
    thread_stats:    Optional[ThreadStats]   = None
    off_heap:        Optional[OffHeapConfig] = None
    heap:            Optional[HeapStats]     = None
    error:           Optional[str]           = None

    @property
    def non_heap_total_mb(self) -> float:
        return round(sum(p.used_mb for p in self.memory_pools), 2)

    @property
    def direct_buffer_mb(self) -> float:
        direct = next((b for b in self.buffer_pools if b.name == "direct"), None)
        return direct.used_mb if direct else 0.0

    @property
    def estimated_total_overhead_mb(self) -> float:
        off_heap_mb = self.off_heap.size_mb if (self.off_heap and self.off_heap.enabled) else 0.0
        thread_mb   = self.thread_stats.estimated_mb if self.thread_stats else 0.0
        return round(
            self.non_heap_total_mb
            + sum(b.used_mb for b in self.buffer_pools)
            + thread_mb
            + off_heap_mb,
            2
        )


# ─────────────────────────────────────────────
# Collector
# ─────────────────────────────────────────────

class JVMMemoryCollector:
    """Reads JVM MBean data and returns an OverheadSnapshot."""

    def __init__(self, spark):
        self._spark = spark

    def collect_driver(self) -> OverheadSnapshot:
        import socket
        sc       = self._spark.sparkContext
        app_name = sc.getConf().get("spark.app.name", "unknown-app")
        app_id   = sc.applicationId
        host     = socket.gethostname()
        ui_url   = sc.uiWebUrl if hasattr(sc, "uiWebUrl") else "unavailable"

        snapshot = OverheadSnapshot(
            label        = "DRIVER",
            timestamp    = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            driver_name  = f"{app_name}  |  app: {app_id}  |  host: {host}",
            executor_url = ui_url,
        )
        try:
            gw  = sc._gateway
            jvm = sc._jvm
            from py4j.java_gateway import java_import
            java_import(gw, "java.lang.management.ManagementFactory")

            self._fill_snapshot(jvm, snapshot)
            self._fill_offheap_config(snapshot)
        except Exception as e:
            snapshot.error = str(e)
        return snapshot

    def collect_executors(self, num_executors: int) -> list[OverheadSnapshot]:
        sc = self._spark.sparkContext

        def _probe(partition_index, _iterator):
            import socket, datetime
            from pyspark import SparkContext
            from py4j.java_gateway import java_import

            host = socket.gethostname()
            try:
                executor_url = f"http://{host}:4040"
            except Exception:
                executor_url = "unavailable"

            snap = {
                "label":           f"executor-{host}",
                "timestamp":       datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
                "host":            host,
                "executor_url":    executor_url,
                "executor_number": partition_index,
                "pools":           [],
                "buffers":         [],
                "threads":         None,
                "error":           None,
            }
            try:
                gw  = SparkContext._gateway
                jvm = SparkContext._jvm
                java_import(gw, "java.lang.management.ManagementFactory")

                for pool in jvm.ManagementFactory.getMemoryPoolMXBeans():
                    if "NON_HEAP" in str(pool.getType()):
                        u = pool.getUsage()
                        snap["pools"].append({
                            "name":         str(pool.getName()),
                            "used_mb":      round(u.getUsed()      / 1024**2, 2),
                            "committed_mb": round(u.getCommitted() / 1024**2, 2),
                            "max_mb":       round(u.getMax()       / 1024**2, 2),
                        })

                for bp in jvm.ManagementFactory.getPlatformMXBeans(
                    jvm.java.lang.management.BufferPoolMXBean
                ):
                    snap["buffers"].append({
                        "name":        str(bp.getName()),
                        "count":       int(bp.getCount()),
                        "used_mb":     round(bp.getMemoryUsed()    / 1024**2, 2),
                        "capacity_mb": round(bp.getTotalCapacity() / 1024**2, 2),
                    })

                th = jvm.ManagementFactory.getThreadMXBean()
                snap["threads"] = {"count": int(th.getThreadCount())}

            except Exception as e:
                snap["error"] = str(e)

            yield snap

        raw_results = (
            sc.range(0, num_executors * 4)
            .repartition(num_executors * 4)
            .mapPartitionsWithIndex(_probe)
            .collect()
        )

        seen, snapshots = set(), []
        for r in raw_results:
            host = r.get("host", "unknown")
            if host in seen:
                continue
            seen.add(host)

            snap = OverheadSnapshot(
                label            = r["label"],
                timestamp        = r["timestamp"],
                executor_number  = r.get("executor_number"),
                executor_url     = r.get("executor_url", "unavailable"),
            )
            if r.get("error"):
                snap.error = r["error"]
            else:
                snap.memory_pools = [MemoryPoolStats(**p) for p in r["pools"]]
                snap.buffer_pools = [BufferPoolStats(**b) for b in r["buffers"]]
                if r.get("threads"):
                    snap.thread_stats = ThreadStats(thread_count=r["threads"]["count"])
            snapshots.append(snap)

        return snapshots

    # ── private helpers ──

    def _fill_snapshot(self, jvm, snapshot: OverheadSnapshot):
        for pool in jvm.ManagementFactory.getMemoryPoolMXBeans():
            if "NON_HEAP" in str(pool.getType()):
                u = pool.getUsage()
                snapshot.memory_pools.append(MemoryPoolStats(
                    name=         str(pool.getName()),
                    used_mb=      round(u.getUsed()      / 1024**2, 2),
                    committed_mb= round(u.getCommitted() / 1024**2, 2),
                    max_mb=       round(u.getMax()       / 1024**2, 2),
                ))

        for bp in jvm.ManagementFactory.getPlatformMXBeans(
            jvm.java.lang.management.BufferPoolMXBean
        ):
            snapshot.buffer_pools.append(BufferPoolStats(
                name=        str(bp.getName()),
                count=       int(bp.getCount()),
                used_mb=     round(bp.getMemoryUsed()    / 1024**2, 2),
                capacity_mb= round(bp.getTotalCapacity() / 1024**2, 2),
            ))

        th = jvm.ManagementFactory.getThreadMXBean()
        snapshot.thread_stats = ThreadStats(thread_count=int(th.getThreadCount()))

        heap = jvm.ManagementFactory.getMemoryMXBean().getHeapMemoryUsage()
        snapshot.heap = HeapStats(
            used_mb=      round(heap.getUsed()      / 1024**2, 2),
            committed_mb= round(heap.getCommitted() / 1024**2, 2),
            max_mb=       round(heap.getMax()       / 1024**2, 2),
        )

    def _fill_offheap_config(self, snapshot: OverheadSnapshot):
        conf    = self._spark.sparkContext.getConf()
        enabled = conf.get("spark.memory.offHeap.enabled", "false").lower() == "true"
        size_b  = conf.get("spark.memory.offHeap.size", "0")
        size_mb = round(int(size_b) / 1024**2, 2) if size_b.isdigit() else 0.0
        snapshot.off_heap = OffHeapConfig(enabled=enabled, size_mb=size_mb)

# Printer
# ─────────────────────────────────────────────

class OverheadMemoryPrinter:
    """Formats and prints OverheadSnapshot objects."""

    SEP  = "=" * 65
    SEP2 = "-" * 65

    def print_snapshot(self, snapshot: OverheadSnapshot, overhead_budget_mb: float = 1024):
        self._print_header(snapshot)
        if snapshot.error:
            print(f"  ERROR: {snapshot.error}")
            return
        self._print_memory_pools(snapshot)
        self._print_buffer_pools(snapshot)
        self._print_threads(snapshot)
        self._print_off_heap(snapshot)
        self._print_heap(snapshot)
        self._print_summary(snapshot, overhead_budget_mb)

    def print_all(self, snapshots: list[OverheadSnapshot], overhead_budget_mb: float = 1024):
        for snap in snapshots:
            self.print_snapshot(snap, overhead_budget_mb)

    # ── private section printers ──

    def _print_header(self, s: OverheadSnapshot):
        print(f"\n{self.SEP}")
        print(f"  OVERHEAD MEMORY BREAKDOWN")
        print(self.SEP)
        print(f"  {'Source':<20}: {s.label}")
        print(f"  {'Timestamp':<20}: {s.timestamp}")

        if s.driver_name:
            print(f"  {'Driver':<20}: {s.driver_name}")
            print(f"  {'Spark UI':<20}: {s.executor_url or 'unavailable'}")
        else:
            num = s.executor_number if s.executor_number is not None else "?"
            print(f"  {'Executor #':<20}: {num}")
            print(f"  {'Executor URL':<20}: {s.executor_url or 'unavailable'}")

        print(self.SEP)

    def _print_memory_pools(self, s: OverheadSnapshot):
        print(f"\n  NON-HEAP POOLS")
        print(f"  {self.SEP2}")
        for p in s.memory_pools:
            max_str = f"{p.max_mb:.1f}" if p.max_mb > 0 else "unlimited"
            print(f"    {p.name:<35} used: {p.used_mb:>8.1f} MB  "
                  f"committed: {p.committed_mb:>8.1f} MB  max: {max_str} MB")
        print(f"    {'TOTAL':<35} used: {s.non_heap_total_mb:>8.1f} MB")

    def _print_buffer_pools(self, s: OverheadSnapshot):
        print(f"\n  BUFFER POOLS (Netty / NIO)")
        print(f"  {self.SEP2}")
        for b in s.buffer_pools:
            print(f"    {b.name:<35} used: {b.used_mb:>8.1f} MB  "
                  f"capacity: {b.capacity_mb:>8.1f} MB  count: {b.count}")

    def _print_threads(self, s: OverheadSnapshot):
        if not s.thread_stats:
            return
        t = s.thread_stats
        print(f"\n  THREAD STACKS")
        print(f"  {self.SEP2}")
        print(f"    {'thread count':<35} {t.thread_count}")
        print(f"    {'stack size each':<35} {t.stack_size_kb} KB")
        print(f"    {'estimated total':<35} {t.estimated_mb:.1f} MB")

    def _print_off_heap(self, s: OverheadSnapshot):
        if not s.off_heap:
            return
        o = s.off_heap
        print(f"\n  SPARK OFF-HEAP CONFIG")
        print(f"  {self.SEP2}")
        print(f"    {'enabled':<35} {o.enabled}")
        print(f"    {'configured size':<35} {o.size_mb:.1f} MB")

    def _print_heap(self, s: OverheadSnapshot):
        if not s.heap:
            return
        h = s.heap
        pct = round(h.used_mb / h.max_mb * 100, 1) if h.max_mb > 0 else 0
        print(f"\n  HEAP SUMMARY")
        print(f"  {self.SEP2}")
        print(f"    {'used / max':<35} {h.used_mb:.1f} MB / {h.max_mb:.1f} MB  ({pct}%)")

    def _print_summary(self, s: OverheadSnapshot, budget_mb: float):
        total     = s.estimated_total_overhead_mb
        remaining = round(budget_mb - total, 1)
        t_mb      = s.thread_stats.estimated_mb if s.thread_stats else 0
        o_mb      = s.off_heap.size_mb if (s.off_heap and s.off_heap.enabled) else 0

        print(f"\n{self.SEP}")
        print(f"  OVERHEAD ESTIMATE SUMMARY")
        print(self.SEP)
        print(f"    {'Non-Heap (Meta + Code + CCS)':<35} {s.non_heap_total_mb:>8.1f} MB")
        print(f"    {'Direct Buffers (Netty/NIO)':<35} {s.direct_buffer_mb:>8.1f} MB")
        print(f"    {'Thread Stacks (estimated)':<35} {t_mb:>8.1f} MB")
        print(f"    {'Spark Off-Heap (configured)':<35} {o_mb:>8.1f} MB")
        print(f"    {self.SEP2}")
        print(f"    {'TOTAL ESTIMATED':<35} {total:>8.1f} MB")
        print(f"    {'Configured Budget':<35} {budget_mb:>8.1f} MB")
        print(f"    {'Remaining Headroom':<35} {remaining:>8.1f} MB")
        print(self.SEP)


# ─────────────────────────────────────────────
# Monitor — main entry point
# ─────────────────────────────────────────────

class SparkOverheadMonitor:
    """
    Main entry point. Wires together collector + printer.

    Usage:
        monitor = SparkOverheadMonitor(spark, overhead_budget_mb=1024)
        monitor.check_driver()
        monitor.check_executors()
        monitor.check_all()

        # Wrap a stage
        df_joined = monitor.run_stage("Big Join", lambda: df1.join(df2, "id"))
    """

    SEP = "=" * 65

    def __init__(self, spark, overhead_budget_mb: float = 1024):
        self._spark     = spark
        self._budget    = overhead_budget_mb
        self._collector = JVMMemoryCollector(spark)
        self._printer   = OverheadMemoryPrinter()

    def check_driver(self) -> OverheadSnapshot:
        snapshot = self._collector.collect_driver()
        self._printer.print_snapshot(snapshot, self._budget)
        return snapshot

    def check_executors(self) -> list[OverheadSnapshot]:
        num_executors = int(
            self._spark.sparkContext.getConf().get("spark.executor.instances", "5")
        )
        print(f"\n  Collecting executor stats ({num_executors} executors)...")
        snapshots = self._collector.collect_executors(num_executors)
        self._printer.print_all(snapshots, self._budget)
        return snapshots

    def check_all(self):
        self.check_driver()
        self.check_executors()

    def run_stage(self, stage_name: str, func):
        """
        Wraps a Spark stage and prints memory after it completes.

        df_joined = monitor.run_stage("Big Join", lambda: df1.join(df2, "id"))
        """
        print(f"\n{self.SEP}")
        print(f"  STARTING STAGE: {stage_name}")
        print(self.SEP)

        result = func()

        print(f"\n  STAGE COMPLETE: {stage_name}")
        self.check_all()

        return result
